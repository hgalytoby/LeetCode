class OrderManagementSystem:

    def __init__(self):
        self.order_hash_map = defaultdict(lambda: defaultdict(set))
        self.order_info = {}

    def addOrder(self, orderId: int, orderType: str, price: int) -> None:
        self.order_info[orderId] = {'id': orderId, 'type': orderType, 'price': price}
        self.order_hash_map[orderType][price].add(orderId)

    def modifyOrder(self, orderId: int, newPrice: int) -> None:
        info = self.order_info[orderId]
        price = info['price']
        self.order_hash_map[info['type']][price].remove(info['id'])
        info['price'] = newPrice
        self.order_hash_map[info['type']][newPrice].add(info['id'])

    def cancelOrder(self, orderId: int) -> None:
        info = self.order_info[orderId]
        price = info['price']
        self.order_hash_map[info['type']][price].remove(info['id'])
        del self.order_info[orderId]
    
    def getOrdersAtPrice(self, orderType: str, price: int) -> List[int]:
        return list(self.order_hash_map[orderType][price])
