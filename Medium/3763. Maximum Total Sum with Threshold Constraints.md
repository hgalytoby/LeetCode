class Solution:
    def maxSum(self, nums: List[int], threshold: List[int]) -> int:
        step = 1
        group = defaultdict(lambda :{"count": 0, "sum": 0})
        result = 0
        for n, t in zip(nums, threshold):
            group[t]["count"] += 1
            group[t]["sum"] += n
        for k in sorted(group.keys()):
            if k > step:
                break
            result += group[k]["sum"]
            step +=  group[k]["count"]
        return result

class Solution:
    def maxSum(self, nums: List[int], threshold: List[int]) -> int:
        step = 1
        queue = []
        result = 0
        for n, t in zip(nums, threshold):
            heappush(queue, (t, -n))
        while queue and queue[0][0] <= step:
            _, n = heappop(queue)
            result += -n
            step += 1
        return result
