```py
class Solution:
    d = [(0, 1), (1, 0), (0, -1), (-1, 0)]

    def numberOfCleanRooms(self, room: List[List[int]]) -> int:
        cur = (0, 0, 0)
        visited = set()
        result = int(room[0][0] == 0)
        n, m = len(room), len(room[0])
        seen = {(0, 0)}
        while cur not in visited:
            x, y, status = cur
            next_x, next_y = x + self.d[status][0], y + self.d[status][1]
            if (next_x, next_y, status) in visited:
                break
            if 0 <= next_x < n and 0 <= next_y < m and room[next_x][next_y] == 0:
                cur = next_x, next_y, status
                visited.add((x, y, status))
                if (next_x, next_y) not in seen:
                    result += 1
                    seen.add((next_x, next_y))
                continue
            visited.add((next_x, next_y, status))
            status = (status + 1) % 4
            cur = x, y, status
        return result
```


### Go
```go
type XY struct {
    X int
    Y int
}

type Cur struct {
    X int
    Y int
    Status int
}

var d = [4]XY{
    {X: 0, Y: 1,},
    {X: 1, Y: 0,},
    {X: 0, Y: -1,},
    {X: -1, Y: 0,},
}

func numberOfCleanRooms(room [][]int) int {
    cur := Cur{X: 0, Y: 0, Status: 0}
    visited := make(map[Cur]struct{})
    result := 0
    n, m := len(room), len(room[0])
    seen := map[XY]struct{}{
        {X: 0, Y: 0}: {},
    }
    
    if room[0][0] == 0 {
        result++
    }

    for {
        if _, ok := visited[cur]; ok {
            break
        }
        nextCur := Cur{
            X: cur.X + d[cur.Status].X,
            Y: cur.Y + d[cur.Status].Y,
            Status: cur.Status,
        }
        if _, ok := visited[nextCur]; ok {
            break
        }
        if 0 <= nextCur.X && nextCur.X < n && 0 <= nextCur.Y && nextCur.Y < m && room[nextCur.X][nextCur.Y] == 0 {
            visited[cur] = struct{}{}
            cur = nextCur
            seenXY := XY{X: nextCur.X, Y: nextCur.Y}
            if _, ok := seen[seenXY]; !ok {
                result++
                seen[seenXY] = struct{}{}
            }
            continue
        }
        visited[nextCur] = struct{}{}
        cur.Status = (cur.Status + 1) % 4
    }
    return result
}
```
