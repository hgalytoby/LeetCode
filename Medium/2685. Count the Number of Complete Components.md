You are given an integer  `n`. There is an  **undirected**  graph with  `n`  vertices, numbered from  `0`  to  `n - 1`. You are given a 2D integer array  `edges`  where  `edges[i] = [ai, bi]`  denotes that there exists an  **undirected**  edge connecting vertices  `ai`  and  `bi`.

Return  _the number of  **complete connected components**  of the graph_.

A  **connected component**  is a subgraph of a graph in which there exists a path between any two vertices, and no vertex of the subgraph shares an edge with a vertex outside of the subgraph.

A connected component is said to be  **complete**  if there exists an edge between every pair of its vertices.

**Example 1:**

**![](https://assets.leetcode.com/uploads/2023/04/11/screenshot-from-2023-04-11-23-31-23.png)**
```
Input: n = 6, edges = [[0,1],[0,2],[1,2],[3,4]]
Output: 3
Explanation: From the picture above, one can see that all of the components of this graph are complete.
```

**Example 2:**

**![](https://assets.leetcode.com/uploads/2023/04/11/screenshot-from-2023-04-11-23-32-00.png)**
```
Input: n = 6, edges = [[0,1],[0,2],[1,2],[3,4],[3,5]]
Output: 1
Explanation: The component containing vertices 0, 1, and 2 is complete since there is an edge between every pair of two vertices. On the other hand, the component containing vertices 3, 4, and 5 is not complete since there is no edge between vertices 4 and 5. Thus, the number of complete components in this graph is 1.
```

**Constraints:**

-   `1 <= n <= 50`
-   `0 <= edges.length <= n * (n - 1) / 2`
-   `edges[i].length == 2`
-   `0 <= ai, bi <= n - 1`
-   `ai != bi`
-   There are no repeated edges.


### Python
```py
class Solution:
    def countCompleteComponents(self, n: int, edges: List[List[int]]) -> int:
        def dfs(j, count):
            visited.add(j)
            counter[j].add(j)
            for k in group[j]:
                if k not in visited:
                    dfs(k, count)
                count[k].add(j)

        group = defaultdict(list)
        for u, v in edges:
            group[u].append(v)
            group[v].append(u)

        visited = set()
        result = 0
        for i in range(n):
            counter = defaultdict(set)
            if i not in visited:
                dfs(i, counter)
                keys = list(counter.keys())
                for idx in range(len(keys) - 1):
                    if counter[keys[idx + 1]] != counter[keys[idx]]:
                        break
                else:
                    result += 1
        return result
```

### C#
```c#
func countCompleteComponents(n int, edges [][]int) int {
    group := make([][]int, n)
	visited := make([]bool, n)
	result := 0

	for _, edge := range edges {
		u, v := edge[0], edge[1]
		group[u] = append(group[u], v)
		group[v] = append(group[v], u)
	}

	var dfs func(int, map[int]map[int]struct{})
	dfs = func(j int, counter map[int]map[int]struct{}) {
		if counter[j] == nil {
			counter[j] = make(map[int]struct{})
		}
		counter[j][j] = struct{}{}
		visited[j] = true

		for _, k := range group[j] {
			if !visited[k] {
				dfs(k, counter)
			}
			if counter[k] == nil {
				counter[k] = make(map[int]struct{})
			}
			counter[k][j] = struct{}{}
		}
	}

	for i := 0; i < n; i++ {
		if !visited[i] {
			counter := make(map[int]map[int]struct{})
			dfs(i, counter)

			keys := make([]int, 0, len(counter))
			for k := range counter {
				keys = append(keys, k)
			}
			isComplete := true
			for idx := 0; idx < len(keys)-1; idx++ {
				if !isEqual(counter[keys[idx]], counter[keys[idx+1]]) {
					isComplete = false
					break
				}
			}
			if isComplete {
				result++
			}
		}
	}
	return result
}

func isEqual(s1, s2 map[int]struct{}) bool {
	if len(s1) != len(s2) {
		return false
	}
	for k := range s1 {
		if _, ok := s2[k]; !ok {
			return false
		}
	}
	return true
}
```
