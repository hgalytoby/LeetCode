class Solution:
    def shareCandies(self, candies: List[int], k: int) -> int:
        if k == 0:
            return len(set(candies))
        counter = Counter(candies)
        count = len(counter)
        result = left = 0
        n = len(candies)
        for right in range(n):
            right_v = candies[right]
            counter[right_v] -= 1
            if counter[right_v] == 0:
                count -= 1
            if right >= k:
                counter[candies[left]] += 1
                if counter[candies[left]] == 1:
                    count += 1
                left += 1
            if right + 1 >= k:
                result = max(result, count)
        return result

            counter[c] += 1
            if counter[c] == 1:
                count += 1

            right_v = candies[right]

                    count += 1
                counter[left_v] += 1
