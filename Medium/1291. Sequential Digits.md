An integer has  _sequential digits_  if and only if each digit in the number is one more than the previous digit.

Return a  **sorted**  list of all the integers in the range  `[low, high]` inclusive that have sequential digits.

**Example 1:**

```
Input: low = 100, high = 300
Output: [123,234]
```

**Example 2:**

```
Input: low = 1000, high = 13000
Output: [1234,2345,3456,4567,5678,6789,12345]
```

**Constraints:**

- `10 <= low <= high <= 10^9`

### Python

```python
class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        low_n = len(str(low))
        high_len = len(str(high))
        result = []
        for i in range(low_n, min(high_len + 1, 10)):
            start = int(''.join(str(i) for i in range(1, i + 1)))
            add = int(''.rjust(i, '1'))
            for j in range(1, 11 - len(str(start))):
                if low <= start <= high:
                    result.append(start)
                start += add
                if start > high:
                    return result
        return result
```

class Solution:
    def sequentialDigits(self, low: int, high: int) -> List[int]:
        sequence = '123456789'
        result = []
        for i in range(len(str(low)), min(len(str(high)) + 1, 10)):
            for j in range(10 - i):
                val = int(sequence[j:j + i])
                if low <= val <= high:
                    result.append(val)
        return result

    def sequentialDigits(self, low: int, high: int) -> List[int]:
        l, h = len(str(low)), len(str(high))
        result = []
        for i in range(l, min(h + 1, 10)):
            for num in self.data[i]:
                if num > high:
                    break
                if num >= low:
                    result.append(num)

func sequentialDigits(low int, high int) []int {

	h := len(strconv.Itoa(high))
	result := make([]int, 0)
	for i := l; i < min(h + 1, 10); i++ {
