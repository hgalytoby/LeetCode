class Solution:
    num_mapping = {
        "z": [("zero", "0")],
        "o": [("one", "1")],
        "t": [("two", "2"), ("three", "3")],
        "f": [("four", "4"), ("five", "5")],
        "s": [("six", "6"), ("seven", "7")],
        "e": [("eight", "8")],
        "n": [("nine", "9")],
    }

    def convertNumber(self, s: str) -> str:
        result = ""
        i = 0
        n = len(s)
        while i < n:
            if s[i] in self.num_mapping:
                for eng, num in self.num_mapping[s[i]]:
                    if s[i : i + len(eng)] == eng:
                        result += num
                        i += len(eng)
                        break
                else:
                    i += 1
            else:
                i += 1
        return result
