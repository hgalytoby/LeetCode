class Solution:
    mapping = {
        "z": ["zero"],
        "o": ["one"],
        "t": ["two", "three"],
        "f": ["four", "five"],
        "s": ["six", "seven"],
        "e": ["eight"],
        "n": ["nine"],
    }
    num_mapping = {
        "zero": "0",
        "one": "1",
        "two": "2",
        "three": "3",
        "four": "4",
        "five": "5",
        "six": "6",
        "seven": "7",
        "eight": "8",
        "nine": "9",
    }

    def convertNumber(self, s: str) -> str:
        result = ""
        i = 0
        n = len(s)
        while i < n:
            if s[i] in self.mapping:
                for eng in self.mapping[s[i]]:
                    if s[i : i + len(eng)] == eng:
                        result += self.num_mapping[eng]
                        i += len(eng)
                        break
                else:
                    i += 1
            else:
                i += 1
        return result
