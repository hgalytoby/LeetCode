MOD = 10**9 + 7

@cache
def helper(n: int, cur: int) -> int:
    v = 1
    for _ in range(n):
        cur += 1
        v = v * cur % MOD
    return v

class Solution:
    def sumOfBlocks(self, n: int) -> int:
        result = cur = 0
        for i in range(1, n + 1):
            v = helper(i, cur)
            cur += i
            result = (result + v) % MOD
        return result
