There is an undirected tree with  `n`  nodes labeled from 1 to  `n`, rooted at node 1. The tree is represented by a 2D
integer array  `edges`  of length  `n - 1`, where  `edges[i] = [ui, vi]`  indicates that there is an edge between
nodes  `ui`  and  `vi`.

Initially, all edges have a weight of 0. You must assign each edge a weight of either  **1**  or  **2**.

The  **cost**  of a path between any two nodes  `u`  and  `v`  is the total weight of all edges in the path connecting
them.

Select any one node  `x`  at the  **maximum**  depth. Return the number of ways to assign edge weights in the path from
node 1 to  `x`  such that its total cost is  **odd**.

Since the answer may be large, return it  **modulo**  `109 + 7`.

**Note:**  Ignore all edges  **not**  in the path from node 1 to  `x`.

**Example 1:**

```
![](https://assets.leetcode.com/uploads/2025/03/23/screenshot-2025-03-24-at-060006.png)
Input:  edges = [[1,2]]
Output:  1
Explanation:
-   The path from Node 1 to Node 2 consists of one edge (`1 → 2`).
-   Assigning weight 1 makes the cost odd, while 2 makes it even. Thus, the number of valid assignments is 1.
```

**Example 2:**

![](https://assets.leetcode.com/uploads/2025/03/23/screenshot-2025-03-24-at-055820.png)

```
Input:  edges = [[1,2],[1,3],[3,4],[3,5]]
Output:  2
Explanation:
-   The maximum depth is 2, with nodes 4 and 5 at the same depth. Either node can be selected for processing.
-   For example, the path from Node 1 to Node 4 consists of two edges (`1 → 3`  and  `3 → 4`).
-   Assigning weights (1,2) or (2,1) results in an odd cost. Thus, the number of valid assignments is 2.
```

**Constraints:**

- `2 <= n <= 105`
- `edges.length == n - 1`
- `edges[i] == [ui, vi]`
- `1 <= ui, vi <= n`
- `edges`  represents a valid tree.

### Python

```python
class Solution:
    modulo = 10 ** 9 + 7

    def assignEdgeWeights(self, edges: List[List[int]]) -> int:
        def build_graph():
            for x, y in edges:
                graph[x].append(y)
                graph[y].append(x)

        graph = defaultdict(list)
        build_graph()
        visited = {1}
        max_depth = 0
        queue = deque([(1, max_depth)])
        while queue:
            edge, depth = queue.popleft()
            max_depth = max(max_depth, depth)
            for neighbor in graph[edge]:
                if neighbor not in visited:
                    visited.add(neighbor)
                    queue.append((neighbor, depth + 1))
        return 2 ** (max_depth - 1) % self.modulo
```

```py
class Solution:
    MOD = 10 ** 9 + 7

    def assignEdgeWeights(self, edges: List[List[int]]) -> int:
        def dfs(val, depth: int = 0):
            nonlocal max_deep
            max_deep = max(max_deep, depth)

            for v in group[val]:
                dfs(v, depth + 1)

        seen = set()
        right_seen = set()
        max_deep = 1
        group = defaultdict(list)
        for left, right in edges:
            seen.add(left)
            seen.add(right)
            right_seen.add(right)
            group[left].append(right)

        root_val = next(iter(seen - right_seen))
        dfs(root_val, 0)
        return 2 ** (max_deep - 1) % self.MOD
```


### Go
```go
var MOD = 1000000007 

func assignEdgeWeights(edges [][]int) int {
    var dfs func(val int, depth int)
    var rootVal int

    result := 1
    seen := make(map[int]struct{})
    rightSeen := make(map[int]struct{})
    maxDeep := 0
    group := make(map[int][]int)

    dfs = func(val int, depth int) {
        maxDeep = max(maxDeep, depth)
        for _, v := range group[val] {
            dfs(v, depth+1)
        }
    }
    for _, edge := range edges {
        left, right := edge[0], edge[1]
        seen[left] = struct{}{}
        seen[right] = struct{}{}
        rightSeen[right] = struct{}{}
        group[left] = append(group[left], right)
    }

    for num := range seen {
        if _, ok := rightSeen[num]; !ok {
            rootVal = num
            break
        }
    }

    dfs(rootVal, 0)

    for i := 0; i < maxDeep - 1; i++ {
        result = result * 2 % MOD
    }
    return result
}
```

### C#
```c#
public class Solution
{
    private const long MOD = 1_000_000_007;

    public int AssignEdgeWeights(int[][] edges)
    {
        int maxDepth = 0;
        int rootVal = 0;
        long result = 1;
        HashSet<int> seen = new HashSet<int>();
        HashSet<int> rightSeen = new HashSet<int>();
        Dictionary<int, List<int>> group = new Dictionary<int, List<int>>();

        foreach (int[] edge in edges)
        {
            int left = edge[0];
            int right = edge[1];

            seen.Add(left);
            seen.Add(right);
            rightSeen.Add(right);

            if (!group.ContainsKey(left))
            {
                group[left] = new List<int>();
            }

            group[left].Add(right);
        }

        foreach (int val in seen)
        {
            if (!rightSeen.Contains(val))
            {
                rootVal = val;
                break;
            }
        }

        void Dfs(int val, int depth)
        {
            maxDepth = Math.Max(maxDepth, depth);
            if (!group.ContainsKey(val))
            {
                return;
            }
            foreach (int v in group[val])
            {
                Dfs(v, depth + 1);
            }
        }

        Dfs(rootVal, 0);
        for (int i = 0; i < maxDepth - 1; i++)
        {
            result = result * 2 % MOD;
        }
        return (int)result;
    }
}
```
