func generateAbbreviations(word string) []string {
    result := []string{}
	n := len(word)
	isDigit := func(s string) bool {
		_, err := strconv.Atoi(s)
		return err == nil
	}

	var backtracking func(cur []string, i int)
	backtracking = func(cur []string, i int) {
		for _, f := range []bool{true, false} {
			arr := make([]string, len(cur))
			copy(arr, cur)
			if f {
				arr = append(arr, string(word[i]))
			} else if len(arr) > 0 && isDigit(arr[len(arr)-1]) {
				num, _ := strconv.Atoi(arr[len(arr)-1])
				arr[len(arr)-1] = strconv.Itoa(num + 1)
			} else {
				arr = append(arr, "1")
			}

			nextI := i + 1

			if nextI == n {
				resStr := ""
				for _, s := range arr {
					resStr += s
				}
				result = append(result, resStr)
			} else {
				backtracking(arr, nextI)
			}
		}
	}

	backtracking([]string{}, 0)
	return result
}

public class Solution {
    public IList<string> GenerateAbbreviations(string word) {
        List<string> result = new List<string>();
        int n = word.Length;

        void Backtracking(List<string> cur, int i) {
            foreach (bool f in new bool[] { true, false }) {
                List<string> arr = new List<string>(cur);
                if (f) {
                    arr.Add(word[i].ToString());
                }
                else if (arr.Count > 0 && int.TryParse(arr[arr.Count - 1], out int num)) {
                    arr[arr.Count - 1] = (num + 1).ToString();
                } else {
                    arr.Add("1");
                }

                int nextI = i + 1;

                if (nextI == n) {
                    result.Add(string.Join("", arr));
                } else {
                    Backtracking(arr, nextI);
                }
            }
        }

        Backtracking(new List<string>(), 0);
        return result;
    }
}
