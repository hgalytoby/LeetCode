You are given an integer array  `nums`  of length  `n`.

You start at index 0, and your goal is to reach index  `n - 1`.

From any index  `i`, you may perform one of the following operations:

-   **Adjacent Step**: Jump to index  `i + 1`  or  `i - 1`, if the index is within bounds.
-   **Prime Teleportation**: If  `nums[i]`  is a  prime number  `p`, you may instantly jump to any index  `j != i`  such that  `nums[j] % p == 0`.

Return the  **minimum**  number of jumps required to reach index  `n - 1`.

**Example 1:**
```
Input:  nums = [1,2,4,6]
Output:  2
Explanation:
One optimal sequence of jumps is:

-   Start at index  `i = 0`. Take an adjacent step to index 1.
-   At index  `i = 1`,  `nums[1] = 2`  is a prime number. Therefore, we teleport to index  `i = 3`  as  `nums[3] = 6`  is divisible by 2.

Thus, the answer is 2.
```

**Example 2:**
```
Input:  nums = [2,3,4,7,9]
Output:  2
Explanation:
One optimal sequence of jumps is:

-   Start at index  `i = 0`. Take an adjacent step to index  `i = 1`.
-   At index  `i = 1`,  `nums[1] = 3`  is a prime number. Therefore, we teleport to index  `i = 4`  since  `nums[4] = 9`  is divisible by 3.

Thus, the answer is 2.
```

**Example 3:**
```
Input:  nums = [4,6,5,8]
Output:  3
Explanation:
-   Since no teleportation is possible, we move through  `0 → 1 → 2 → 3`. Thus, the answer is 3.
```

**Constraints:**

-   `1 <= n == nums.length <= 105`
-   `1 <= nums[i] <= 106`


### Python
```py
LIMIT = 10 ** 6 + 1
TP = [[] for _ in range(LIMIT)]
for i in range(2, LIMIT):
    for j in range(i, LIMIT, i):
        TP[j].append(i)

class Solution:
    def minJumps(self, nums: List[int]) -> int:
        index = defaultdict(list)
        n = len(nums)
        memo = [n + 1] * n
        result = math.inf
        queue = deque([(nums[0], 0, 0)])
        group = defaultdict(list)
        for i, num in enumerate(nums):
            index[num].append(i)
        for num, arr in index.items():
            if len(TP[num]) > 1:
                for v in TP[num]:
                    if v in index and num != v:
                        group[v].append(num)
            elif len(TP[num]) == 1:
                group[num].append(num)
        while queue:
            num, i, step = queue.popleft()
            if i == n - 1:
                result = min(result, step)
                continue
            if memo[i] < step:
                continue
            next_step = step + 1
            for nxt_i in [i + 1, i - 1]:
                if 0 <= nxt_i < n and memo[nxt_i] > next_step:
                    memo[nxt_i] = next_step
                    queue.append((nums[nxt_i], nxt_i, next_step))
            if len(TP[num]) == 1:
                while group[num]:
                    v = group[num].pop()
                    for j in index[v]:
                        if memo[j] > next_step:
                            queue.append((v, j, next_step))
                            memo[j] = next_step
        return result
```

### Go
```go
const limit = 1_000_000 + 1

var tp = make([][]int, limit)

type State struct {
	I    int
	Step int
}

func init() {
	for i := 2; i < limit; i++ {
		for j := i; j < limit; j += i {
			tp[j] = append(tp[j], i)
		}
	}
}

func minJumps(nums []int) int {
	n := len(nums)
	memo := make([]int, n)
	for i := range memo {
		memo[i] = n + 1
	}
	result := n
	queue := []State{{I: 0, Step: 0}}
	group := make(map[int][]int)
	index := make(map[int][]int)
	for i, num := range nums {
		index[num] = append(index[num], i)
	}
	for num := range index {
		if len(tp[num]) > 1 {
			for _, v := range tp[num] {
				if index[v] != nil && num != v {
					group[v] = append(group[v], num)
				}
			}
		} else if len(tp[num]) == 1 {
			group[num] = append(group[num], num)
		}
	}
    head := 0
	for head < len(queue) {
		state := queue[head]
        head++
		i, step := state.I, state.Step
        num := nums[i]
		if i == n-1 {
			result = min(result, step)
			continue
		}
		if memo[i] < step {
			continue
		}
		nextStep := step + 1
		for _, nxtI := range [...]int{i + 1, i - 1} {
			if 0 <= nxtI && nxtI < n && memo[nxtI] > nextStep {
				memo[nxtI] = nextStep
				queue = append(queue, State{I: nxtI, Step: nextStep})
			}
		}
		if len(tp[num]) == 1 {
			for len(group[num]) > 0 {
				v := group[num][0]
				group[num] = group[num][1:]
				for _, j := range index[v] {
					if memo[j] > nextStep {
						queue = append(queue, State{I: j, Step: nextStep})
						memo[j] = nextStep
					}
				}
			}
		}
	}
	return result
}
```
