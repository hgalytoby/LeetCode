# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def levelMedian(self, root: Optional[TreeNode], level: int) -> int:
        def dfs(node, deep: int = 0):
            if not node or deep > level:
                return
            if deep == level:
                arr.append(node.val)
            dfs(node.left, deep + 1)
            dfs(node.right, deep + 1)
            
        arr = []
        dfs(root)
        n = len(arr)
        x, y = divmod(n, 2)
        if not arr:
            return -1
        if y == 0:
            return max(arr[x], arr[x - 1])
        return arr[x]
