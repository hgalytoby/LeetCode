class ValidWordAbbr:

    def __init__(self, dictionary: List[str]):
        self.unique = Counter()
        self.word = set()
        for d in dictionary:
            if d not in self.word:
                self.unique[self.helper(d)] += 1
            self.word.add(d)

    def helper(self, word: str) -> str:
        n = len(word)
        return word if n <= 2 else f"{word[0]}{n - 2}{word[-1]}"

    def isUnique(self, word: str) -> bool:
        w = self.helper(word)
        return word in self.word and self.unique[w] == 1 or w not in self.unique

public class ValidWordAbbr {
    private readonly Dictionary<string, int> Unique = new();
    private readonly HashSet<string> Word = new();

    public ValidWordAbbr(string[] dictionary) {
        foreach (string d in dictionary)
        {
            if (!Word.Contains(d))
            {
                string w = Helper(d);
                if (!Unique.ContainsKey(w))
                {
                    Unique[w] = 0;
                }
                Unique[w]++;
            }
            Word.Add(d);
        }
    }

    public string Helper(string word) {
        int n = word.Length;
        if (n <= 2)
        {
            return word;
        }
        return $"{word[0].ToString()}{n.ToString()}{word[^1].ToString()}";
    }
    
    public bool IsUnique(string word) {
        string w = Helper(word);
        return Word.Contains(word) && Unique[w] == 1 || !Unique.ContainsKey(w);
    }
}
