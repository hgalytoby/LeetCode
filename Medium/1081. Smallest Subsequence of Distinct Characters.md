Given a string  `s`, return  _the_ _lexicographically smallest_  _subsequence_ _of_  `s`  _that contains all the distinct characters of_  `s`  _exactly once_.

**Example 1:**
```
Input: s = "bcabc"
Output: "abc"
```

**Example 2:**
```
Input: s = "cbacdcbc"
Output: "acdb"
```

**Constraints:**

-   `1 <= s.length <= 1000`
-   `s`  consists of lowercase English letters.

**Note:** This question is the same as 316: [https://leetcode.com/problems/remove-duplicate-letters/](https://leetcode.com/problems/remove-duplicate-letters/)


### Python
```py
class Solution:
    def smallestSubsequence(self, s: str) -> str:
        counter = Counter(s)
        stack = []
        for c in s:
            counter[c] -= 1
            if c not in stack:
                while stack and counter[stack[-1]] > 0 and c < stack[-1]:
                    stack.pop()
                stack.append(c)
        return ''.join(stack)
```


### Go
```go
func smallestSubsequence(s string) string {
    count := [26]int{}
    inStack := [26]bool{}
    stack := []rune{}
    for i := 0; i < len(s); i++ {
        count[s[i]-'a']++
    }
    for _, c := range s {
        count[c-'a']--
        if inStack[c-'a'] {
            continue
        }
        for len(stack) > 0 && c < stack[len(stack)-1] && count[stack[len(stack)-1]-'a'] > 0 {
            top := stack[len(stack)-1]
            stack = stack[:len(stack)-1]
            inStack[top-'a'] = false
        }
        stack = append(stack, c)
        inStack[c-'a'] = true
    }
    return string(stack)
}
```

### C#
```c#
public class Solution {
    public string SmallestSubsequence(string s) {
        int[] count = new int[26];
        foreach (char c in s) {
            count[c - 'a']++;
        }
        bool[] inStack = new bool[26];
        Stack<char> stack = new Stack<char>();
        foreach (char c in s) {
            count[c - 'a']--;
            if (inStack[c - 'a']) {
                continue;
            }
            while (stack.Count > 0 && c < stack.Peek() && count[stack.Peek() - 'a'] > 0) {
                char top = stack.Pop();
                inStack[top - 'a'] = false;
            }
            stack.Push(c);
            inStack[c - 'a'] = true;
        }
        char[] resultArray = stack.ToArray();
        Array.Reverse(resultArray);
        return new string(resultArray);
    }
}
```
