class Solution:
    def sortArray(self, nums: List[int], pre: List[int]) -> int:
        target = sorted(nums)
        if nums == target:
            return 0
        visited = {tuple(nums)}
        queue = deque([[nums, 0]])
        while queue:
            arr, count = queue.popleft()
            for p in pre:
                new_arr = arr[:p][::-1] + arr[p:]
                if new_arr == target:
                    return count + 1
                k = tuple(new_arr)
                if k in visited:
                    continue
                visited.add(k)
                queue.append([new_arr, count + 1])
        return -1
