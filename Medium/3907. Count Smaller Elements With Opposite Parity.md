class Solution:
    def countSmallerOppositeParity(self, nums: list[int]) -> list[int]:
        n = len(nums)
        result = [0] * n
        evens, odds = SortedList(), SortedList()
        for i in range(n - 1, -1, -1):
            val = nums[i]
            if val % 2 == 1:
                result[i] = evens.bisect_left(val)
                odds.add(val)
            else:
                result[i] = odds.bisect_left(val)
                evens.add(val)
        return result
