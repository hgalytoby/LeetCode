class Solution:
    def maxSameLengthRuns(self, s: str) -> int:
        cur = []
        group = defaultdict(int)
        for c in s:
            if not cur or cur[-1] == c:
                cur.append(c)
            else:
                group[len(cur)] += 1
                cur = [c]
        if cur:
            group[len(cur)] += 1
        return max(group.values())
