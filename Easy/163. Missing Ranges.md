class Solution:
    def findMissingRanges(self, nums: List[int], lower: int, upper: int) -> List[List[int]]:
        if not nums:
            return [[lower, upper]]
        result = []
        n = len(nums)
        if lower != nums[0]:
            result.append([lower, nums[0] - 1])
        result.extend([[nums[i - 1] + 1, nums[i] - 1] for i in range(1, n) if nums[i] - nums[i - 1] > 1])
        if upper != nums[-1]:
            result.append([nums[-1] + 1, upper])
        return result
