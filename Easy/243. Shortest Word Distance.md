class Solution:
    def shortestDistance(self, wordsDict: List[str], word1: str, word2: str) -> int:
        w1_i = w2_i = -1
        result = math.inf
        for i, w in enumerate(wordsDict):
            if w == word1:
                w1_i = i
            elif w == word2:
                w2_i = i
            if w1_i != -1 and w2_i != -1:
                result = min(result, abs(w1_i - w2_i))
        return result
