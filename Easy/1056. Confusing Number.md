Premium


### Python
```py
class Solution:
    BAN = {'2', '3', '4', '5', '7'}
    MAPPING = {'0': '0', '1': '1', '6': '9', '8': '8', '9': '6'}

    def confusingNumber(self, n: int) -> bool:
        v = ''
        for c in str(n)[::-1]:
            if c in self.BAN:
                return False
            v += self.MAPPING[c]
        return n != int(v)
```

var BAN = map[rune]struct{}{
    '2': {},
    '3': {},
    '4': {},
    '5': {},
    '7': {},
}
var MAPPING = map[rune]string{
    '0': "0",
    '1': "1",
    '6': "9",
    '8': "8",
    '9': "6",
}

func confusingNumber(n int) bool {
    v := ""
    r := []rune(strconv.Itoa(n))
	slices.Reverse(r)
    for _, c := range r {
        if _, ok := BAN[c]; ok {
            return false
        }
        v += MAPPING[c]
    }
    val, _ := strconv.Atoi(v)
    return n != val
}
