### Python
```py
class Solution:
    def fixedPoint(self, arr: List[int]) -> int:
        left, right = 0, len(arr) - 1
        result = -1
        while left <= right:
            mid = (left + right) // 2
            if arr[mid] >= mid:
                if arr[mid] == mid:
                    result = mid
                right = mid - 1
            else:
                left = mid + 1
        return result
```
