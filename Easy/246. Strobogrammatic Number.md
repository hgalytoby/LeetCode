```py
mapping = {"1": "1", "6": "9", "8": "8", "9": "6", "0": "0"}


class Solution:
    def isStrobogrammatic(self, num: str) -> bool:
        return not any(
            num[i] not in mapping
            or num[~i] not in mapping
            or mapping[num[i]] != num[~i]
            for i in range(sum(divmod(len(num), 2)))
        )
```
