Given an array of integers  `arr`, you are initially positioned at the first index of the array.

In one step you can jump from index  `i`  to index:

-   `i + 1`  where: `i + 1 < arr.length`.
-   `i - 1`  where: `i - 1 >= 0`.
-   `j`  where:  `arr[i] == arr[j]`  and  `i != j`.

Return  _the minimum number of steps_  to reach the  **last index**  of the array.

Notice that you can not jump outside of the array at any time.

**Example 1:**
```
Input: arr = [100,-23,-23,404,100,23,23,23,3,404]
Output: 3
Explanation: You need three jumps from index 0 --> 4 --> 3 --> 9. Note that index 9 is the last index of the array.
```

**Example 2:**
```
Input: arr = [7]
Output: 0
Explanation: Start index is the last index. You do not need to jump.
```

**Example 3:**
```
Input: arr = [7,6,9,6,9,6,9,7]
Output: 1
Explanation: You can jump directly from index 0 to index 7 which is last index of the array.
```

**Constraints:**

-   `1 <= arr.length <= 5 * 104`
-   `-108 <= arr[i] <= 108`


### Python
```py
class Solution:
    def minJumps(self, arr: List[int]) -> int:
        tp = defaultdict(list)
        for i, n in enumerate(arr):
            tp[n].append(i)
        n = len(arr)
        queue = deque([(0, 0)])
        result = target = n -1
        memo = {0: n}
        while queue:
            cur_i, count = queue.popleft()
            if cur_i == target:
                result = min(result, count)
                break
            num = arr[cur_i]
            next_count = count + 1
            for next_i in tp[num]:
                if memo.get(next_i, n) > next_count:
                    queue.append((next_i, next_count))
                    memo[next_i] = next_count
            tp[num].clear()
            for move in [-1, 1]:
                next_i = cur_i + move
                if 0 <= next_i < n and memo.get(next_i, n) > next_count:
                    queue.append((next_i, next_count))
                    memo[next_i] = next_count
        return result
```


### Go
```go
type QueueItem struct {
    Cur int
    Count int
}

func minJumps(arr []int) int {
    tp := make(map[int][]int)
    for i, n := range arr {
        tp[n] = append(tp[n], i)
    }
    n := len(arr)
    queue := []QueueItem{{Cur: 0, Count: 0}}
    result, target := n - 1, n - 1
    memo := make([]int, n)
    for i := range memo {
        memo[i] = n
    }
    head := 0
    for head < len(queue) {
        curI, count := queue[head].Cur, queue[head].Count
        head++
        if curI == target {
            return count
        }
        num := arr[curI]
        nextCount := count + 1
        for _, nextI := range tp[num] {
            if memo[nextI] > nextCount {
                queue = append(queue, QueueItem{Cur: nextI, Count: nextCount})
                memo[nextI] = nextCount
            }
        }
        tp[num] = nil
        for _, move := range [2]int{-1, 1} {
            nextI := curI + move
            if 0 <= nextI && nextI < n && memo[nextI] > nextCount {
                queue = append(queue, QueueItem{Cur: nextI, Count: nextCount})
                memo[nextI] = nextCount
            } 
        }   
    }
    return result
}
```
